Get Playerdata

Get Player​

local player = exports['wc_core']:getPlayer(source)
  • return: table
    • value: number (source)

Example:

local player = exports['wc_core']:getPlayer(source)

Get Player With Character Identifier​

local source, player = exports['wc_core']:getPlayer(char_identifier)
  • return: source: number or string "offline"
  • return: player: table
    • value: string

Get All Players By Filter​

local players = exports['wc_core']:getAllPlayers(key, value)
  • return: table
    • key: string
    • value: string or table

Example:

local policeOfficers = exports['wc_core']:getAllPlayers('job', 'police')
local multipleJobs = exports['wc_core']:getAllPlayers('job', {'police', 'ambulance'})

Get Active Players By Filter​

local players = exports['wc_core']:getPlayers(key, value)
  • return: table
    • key: string
    • value: string or table

Example:

local policeOfficers = exports['wc_core']:getPlayers('job', 'police')
local multipleJobs = exports['wc_core']:getPlayers('job', {'police', 'ambulance'})

Player​

Get player license​

Player identifier / rockstar license

player.player_license
  • return: license:xxxxxx

Get player groups​

Player groups

player.player_groups
  • return: table
    • Player groups such as player, admin etc

Character​

Get character identifier​

Character identifier

player.char_identifier
  • return: string
    • Number as a string

Get character identity​

Character identifier

player.char_identity
  • return: table
    • firstname: string
    • lastname: string
    • dateofbirth: string
    • sex: string

Example:

local firstname = player.char_identity.firstname
local dateofbirth = player.char_identity.dateofbirth

Get character vitality​

Character vitality

player.char_vitality
  • return: table
    • health: number
    • armor: number
    • hunger: number
    • thirst: number

Example:

local health = player.char_vitality.health
local hunger = player.char_vitality.hunger

Get character metadata​

Character metadata

player.char_metadata
  • return: table
